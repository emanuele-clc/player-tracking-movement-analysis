# CPU-only PyTorch wheel index - keeps installs light and avoids pulling CUDA
# drivers on machines/hosts without a GPU (e.g. Hugging Face Spaces free CPU
# tier, or a laptop). pip still prefers the default index when a matching
# wheel exists there, so this is safe to leave in even with a GPU present.
--extra-index-url https://download.pytorch.org/whl/cpu

# Data access
SoccerNet
roboflow                # download football-specific YOLOv8 weights (player/ball/pitch-keypoint
                        # detection) from Roboflow Universe - free API key, see data/README.md

# Computer vision: detection + tracking
ultralytics             # YOLOv8 player/ball detection (installs torch - the --extra-index-url
                        # above steers this to the CPU wheel automatically)
opencv-python-headless   # headless build - no GUI/X11 deps, works on servers/Spaces; same API
supervision<0.28        # pinned below 0.28: ByteTrack API (update_with_detections) is
                        # deprecated from 0.28 onward in favor of the separate `trackers`
                        # package, which isn't stable enough yet to switch to
norfair                 # alternative lightweight tracker, useful as a fallback/comparison

# Analysis
numpy
pandas
pyarrow                 # parquet read/write for the tracklet/tracking datasets
scipy
scikit-learn      # clustering (k-means, GMM, DBSCAN)

matplotlib
plotly
tqdm
